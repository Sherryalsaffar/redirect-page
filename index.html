async function redirect() { try { const webhookUrl = 'https://hooks.zapier.com/hooks/catch/YOUR_WEBHOOK_ID/'; console.log("Starting redirect function"); // Log script start console.log("Webhook URL being fetched:", webhookUrl); const response = await fetch(webhookUrl, { method: 'POST', }); console.log("Raw response:", response); const data = await response.json(); console.log("Parsed JSON response:", data); if (data.selectedURL) { console.log("Redirecting to:", data.selectedURL); window.location.href = data.selectedURL; } else { console.error('No redirect URL found in response:', data); } } catch (error) { console.error('Error during fetch/redirect:', error); } } window.onload = redirect;
